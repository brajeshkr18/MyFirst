CREATE PROCEDURE [dbo].[Web_SaveOnCallPeople] 
@id bigint = null,
@OnCAllPersonID varchar(250),
@FirstName varchar(250),
@LastName varchar(250),
@PhoneNumber decimal,
@CellNO decimal,
@EmailAddress varchar(50) ,
@PracticeId bigint,
@SpecialityId bigint,
@IsEmtala bit,
@IsCredentialed bit,
@CanRequest bit,
@CallCarrier varchar(50),
@UserId varchar(50),
@Picture varchar(max)



AS
BEGIN
SET NOCOUNT OFF;
   if exists(select * from tblOnCallPeople usr where usr.id = @id and usr.Isactive = 1 )
   BEGIN
   		  update tblOnCallPeople
		  set
		  FirstName = @FirstName,
		  LastName = @LastName,
		  PhoneNumber = @PhoneNumber,
		  CellNO=@CellNO,
		  EmailAddress=@EmailAddress,
		  PracticeId=@PracticeId,
		  SpecialityId =@SpecialityId,
		  IsEmtala = @IsEmtala,
		  IsCredentialed = @IsCredentialed,
		  CanRequest=@CanRequest,
		  CallCarrier=@CallCarrier,
		  modifiedby = @UserId,
		  modifiedDate = GetDate(),
		  CreatedDate=GetDate(),
		  CreatedBy=@UserId	 ,
		  FullName= @FirstName+' '+@LastName ,
		  Picture = @Picture
		  where id=@id
		  select 2 as resultoncall

					return

  END
  IF NOT EXISTS(select * from tblOnCallPeople usr where usr.EmailAddress = @EmailAddress and usr.Isactive = 1)
	BEGIN
			insert into tblOnCallPeople (FirstName,LastName,PhoneNumber,CellNO, EmailAddress,PracticeId,SpecialityId,IsEmtala,IsCredentialed, IsActive,CanRequest,CallCarrier, CreatedDate,modifiedDate,CreatedBy,modifiedby,FullName,Picture)
			values (@FirstName,@LastName,@PhoneNumber ,@CellNO,@EmailAddress,@PracticeId,@SpecialityId,@IsEmtala,@IsCredentialed, 1,@CanRequest,@CallCarrier, GetDate(),GetDate(),@UserId,@UserId,@FirstName+' '+@LastName,@Picture )
			select 1 as resultoncall
			return
	END
	ELSE
	BEGIN
	select 0 as resultoncall

	END
End

CREATE PROCEDURE [dbo].[Web_SaveSpeciality] 

@Id BIGINT = NULL,
@Speciality VARCHAR(150),
@Description VARCHAR(100)
,@IsActive BIT
,@CreatedBy BIGINT
,@ModifiedBy BIGINT

AS
BEGIN

SET NOCOUNT OFF;
	IF EXISTS (select * from tblSpeciality where Id = @Id)
		BEGIN
			Update tblSpeciality set
			Speciality = @Speciality
			,Description = @Description
			,IsActive = @IsActive
			,ModifiedDate = GETDATE()
			,ModifiedBy = @ModifiedBy
			where Id = @Id
			select 2 As SpecialityResult
			return
		END

	IF NOT EXISTS(select * from tblSpeciality where Speciality = @Speciality and isactive = 1)
		BEGIN
		
			Insert into tblSpeciality
			(Speciality,Description,IsActive,CreatedDate,ModifiedDate,CreatedBy,ModifiedBy)
			Values
			(@Speciality,@Description,@IsActive,GetDate(),GetDate(),@CreatedBy,@ModifiedBy)
			
			select 1  as SpecialityResult
			return
		END
	ELSE
		BEGIN
			select 0 SpecialityResult
			return
		END
	
END

CREATE PROCEDURE [dbo].[Web_Users]
 @Id BIGINT = 0
AS
BEGIN
	
	SET NOCOUNT ON;
	IF(@Id = 0)
	BEGIN
	select * from tblUsers  t inner join mtblRoles m on m.id =t.roleid  where t.isactive =1
	end
	ELSE
	BEGIN
		select * from tblUsers  t inner join mtblRoles m on m.id =t.roleid  where t.Id = @Id and t.isactive = 1
	END
  
END
